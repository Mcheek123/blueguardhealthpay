name: Deploy to Cloud Run

on:
  push:
    branches:
      - main

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout code
      uses: actions/checkout@v2

    - name: Set up Google Cloud SDK
      uses: google-github-actions/setup-gcloud@v0.2.1
      with:
        project_id: ${{ secrets.GCP_PROJECT }}
        service_account_key: ${{ secrets.GCP_SA_KEY }}

    - name: Build and push Docker image
      run: |
        gcloud auth configure-docker
        docker build -t gcr.io/${{ secrets.GCP_PROJECT }}/blueguard-avatar:latest .
        docker push gcr.io/${{ secrets.GCP_PROJECT }}/blueguard-avatar:latest

    - name: Deploy to Cloud Run
      run: |
        gcloud run deploy blueguard-avatar \
          --image gcr.io/${{ secrets.GCP_PROJECT }}/blueguard-avatar:latest \
          --region us-east4 \
          --platform managed \
          --allow-unauthenticated
